1//! Lightweight C/C++ parser using [`tree-sitter-cpp`].
2//!
3//! This module provides a fast, zero‑dependency alternative to Clang for
4//! extracting call‑graph facts from C and C++ source code.  It does not
5//! perform semantic analysis (types, overload resolution); it only produces
6//! syntactic nodes suitable for building a call graph.
7//!
8//! # Supported node kinds
9//!
10//! * `Function` – function definitions and declarations,
11//! * `Class` – class/struct definitions,
12//! * `CallSite` – call expressions,
13//! * `Variable` – variable declarations outside parameter lists,
14//! * `Parameter` – parameter declarations inside parameter lists.
15//!
16//! # Example
17//!
18//! ```rust
19//! use icb_parser::cpp_tree_sitter::parse_cpp_file;
20//!
21//! let code = r#"
22//!     int add(int a, int b) { return a + b; }
23//!     void main() { add(1, 2); }
24//! "#;
25//! let facts = parse_cpp_file(code).unwrap();
26//! assert!(facts.iter().any(|n| n.kind == icb_common::NodeKind::Function));
27//! assert!(facts.iter().any(|n| n.kind == icb_common::NodeKind::CallSite));
28//! ```
29
30use icb_common::{IcbError, Language, NodeKind};
31use tree_sitter::{Node, Parser};
32
33use crate::facts::RawNode;

35/// Parse a C/C++ source file and return a flat list of facts.
36///
37/// # Errors
38///
39/// Returns [`IcbError::Parse`] if the tree‑sitter parser cannot be
40/// initialised or the source contains syntax errors.
41pub fn parse_cpp_file(source: &str) -> Result<Vec<RawNode>, IcbError> {
42    let mut parser = Parser::new();
43    parser
44        .set_language(&tree_sitter_cpp::language())
45        .map_err(|e| IcbError::Parse(format!("cannot set tree-sitter-cpp language: {e}")))?;
46
47    let tree = parser
48        .parse(source, None)
49        .ok_or_else(|| IcbError::Parse("tree-sitter parse returned None".into()))?;
50
51    let mut facts = Vec::new();
52    traverse_node(tree.root_node(), source, &mut facts, None);
53    Ok(facts)
54}

56/// Recursively walk the CST and push relevant nodes into `facts`.
57///
58/// Returns the index of the last node that should serve as parent for
59/// subsequent siblings.
60fn traverse_node(
61    node: Node,
62    source: &str,
63    facts: &mut Vec<RawNode>,
64    parent_idx: Option<usize>,
65) -> Option<usize> {
66    let kind = node.kind();
67
68    let (node_kind, name, is_container) = match kind {
69        "function_definition" | "function_declaration" => {
70            let name = function_name(node, source).unwrap_or_default();
71            (NodeKind::Function, Some(name), true)
72        }
73        "class_specifier" | "struct_specifier" => {
74            let name = node
75                .child_by_field_name("name")
76                .map(|n| {
77                    n.utf8_text(source.as_bytes())
78                        .unwrap_or_default()
79                        .to_string()
80                })
81                .unwrap_or_default();
82            (NodeKind::Class, Some(name), true)
83        }
84        "call_expression" => {
85            let name = node
86                .child_by_field_name("function")
87                .map(|n| {
88                    n.utf8_text(source.as_bytes())
89                        .unwrap_or_default()
90                        .to_string()
91                })
92                .unwrap_or_default();
93            (NodeKind::CallSite, Some(name), false)
94        }
95        "declaration" => {
96            let name = node
97                .child_by_field_name("declarator")
98                .or_else(|| node.child_by_field_name("name"))
99                .map(|n| {
100                    n.utf8_text(source.as_bytes())
101                        .unwrap_or_default()
102                        .to_string()
103                })
104                .unwrap_or_default();
105            if parent_kind_is(node, "parameter_list") {
106                (NodeKind::Parameter, Some(name), false)
107            } else {
108                (NodeKind::Variable, Some(name), false)
109            }
110        }
111        _ => {
112            let mut current_parent = parent_idx;
113            for child in node.children(&mut node.walk()) {
114                current_parent = traverse_node(child, source, facts, current_parent);
115            }
116            return current_parent;
117        }
118    };
119
120    let start = node.start_position();
121    let end = node.end_position();
122
123    let idx = facts.len();
124    facts.push(RawNode {
125        language: Language::CppTreeSitter,
126        kind: node_kind,
127        name,
128        usr: None,
129        start_line: start.row + 1,
130        start_col: start.column,
131        end_line: end.row + 1,
132        end_col: end.column,
133        children: Vec::new(),
134        source_file: None,
135    });
136
137    if let Some(pidx) = parent_idx {
138        facts[pidx].children.push(idx);
139    }
140
141    if is_container {
142        let new_parent = Some(idx);
143        let mut current_parent = new_parent;
144        for child in node.children(&mut node.walk()) {
145            current_parent = traverse_node(child, source, facts, current_parent);
146        }
147        new_parent
148    } else {
149        parent_idx
150    }
151}

153/// Extract the function name from a function definition/declaration node.
154fn function_name(node: Node, source: &str) -> Option<String> {
155    // Walk the declarator chain: function_definition → declarator (function_declarator) → declarator (identifier)
156    if let Some(decl) = node.child_by_field_name("declarator") {
157        if decl.kind() == "function_declarator" {
158            if let Some(name_node) = decl.child_by_field_name("declarator") {
159                return Some(name_node.utf8_text(source.as_bytes()).ok()?.to_string());
160            }
161            // If the function_declarator has no named 'declarator' child, fall back to its text.
162            // e.g. operator overloads may have a different structure.
163            return Some(decl.utf8_text(source.as_bytes()).ok()?.to_string());
164        }
165        // For plain function declarations the declarator field may point directly to an identifier.
166        return Some(decl.utf8_text(source.as_bytes()).ok()?.to_string());
167    }
168    // Last resort: take the first identifier child.
169    let mut cursor = node.walk();
170    for child in node.children(&mut cursor) {
171        if child.kind() == "identifier" {
172            return child
173                .utf8_text(source.as_bytes())
174                .ok()
175                .map(|s| s.to_string());
176        }
177    }
178    None
179}

181/// Check whether the node's immediate parent has the expected kind.
182fn parent_kind_is(node: Node, expected: &str) -> bool {
183    node.parent().is_some_and(|p| p.kind() == expected)
184}

186#[cfg(test)]
187mod tests {
188    use super::*;
189
190    #[test]
191    fn parse_simple_function() {
192        let facts = parse_cpp_file("void foo() {}").unwrap();
193        assert_eq!(facts.len(), 1);
194        assert_eq!(facts[0].kind, NodeKind::Function);
195        assert_eq!(facts[0].name.as_deref(), Some("foo"));
196    }
197
198    #[test]
199    fn parse_function_with_call() {
200        let code = "void bar() {} void baz() { bar(); }";
201        let facts = parse_cpp_file(code).unwrap();
202        let calls: Vec<_> = facts
203            .iter()
204            .filter(|n| n.kind == NodeKind::CallSite)
205            .collect();
206        assert!(!calls.is_empty());
207    }
