8/// Central entry point for parsing source code.
9///
10/// Supports single files and whole directories (parallel parsing).
11/// Currently Python is supported out of the box.
12/// For C/C++ support use the `icb-clang` crate directly
13/// or via the CLI subcommand.
14#[derive(Default)]
15pub struct ParserManager;
16
17impl ParserManager {
18    pub fn new() -> Self {
19        Self
20    }
21
22    /// Parse a single source file into a flat list of facts.
23    pub fn parse_file(&self, lang: Language, source: &str) -> Result<Vec<RawNode>, IcbError> {
24        match lang {
25            Language::Python => parse_python(source),
26            _ => Err(IcbError::Parse(format!("unsupported language {:?}", lang))),
27        }
28    }
29
30    /// Parse all source files inside a directory (recursively).
31    ///
32    /// Returns a vector of `(relative_path, Vec<RawNode>)` for each
33    /// successfully parsed file. Files that fail to parse are skipped with
34    /// a warning.
35    ///
36    /// # Supported extensions
37    /// - Python: `.py`
38    /// - C/C++: `.cpp`, `.cxx`, `.c`, `.h`, `.hpp` (via Clang, if `icb-clang` is used separately)
39    pub fn parse_directory(
40        &self,
41        lang: Language,
42        root: &Path,
43    ) -> Result<Vec<(String, Vec<RawNode>)>, IcbError> {
44        let mut files = Vec::new();
45        collect_files(root, &mut files, lang)?;
46
47        let results: Vec<_> = files
48            .into_par_iter()
49            .filter_map(|path| {
50                let content = fs::read_to_string(&path).ok()?;
51                match self.parse_file(lang, &content) {
52                    Ok(facts) => {
53                        let relative = path.strip_prefix(root).unwrap_or(&path);
54                        Some((relative.display().to_string(), facts))
55                    }
56                    Err(e) => {
57                        log::warn!("Skipping {}: {}", path.display(), e);
58                        None
59                    }
60                }
61            })
62            .collect();
63        Ok(results)
64    }
65}
66
67/// Recursively collect source files with the appropriate extension for the language.
68fn collect_files(
69    dir: &Path,
70    files: &mut Vec<std::path::PathBuf>,
71    lang: Language,
72) -> Result<(), IcbError> {
73    let extensions: &[&str] = match lang {
74        Language::Python => &["py"],
75        Language::Rust => &["rs"],
76        Language::JavaScript => &["js"],
77        Language::Cpp => &["cpp", "cxx", "c", "h", "hpp"],
78    };
79    for entry in fs::read_dir(dir).map_err(IcbError::Io)? {
80        let entry = entry.map_err(IcbError::Io)?;
81        let path = entry.path();
82        if path.is_dir() {
83            collect_files(&path, files, lang)?;
84        } else if let Some(ext) = path.extension().and_then(|s| s.to_str()) {
85            if extensions.contains(&ext) {
86                files.push(path);
87            }
88        }
89    }
90    Ok(())
91}
