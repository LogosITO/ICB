1//! Graph construction and caching logic for the server.
2//!
3//! # Overview
4//!
5//! This module is responsible for turning raw parser facts into a fully
6//! resolved [`CodePropertyGraph`].  It supports three distinct workflows:
7//!
8//! * **build** – parse a project or a single file and construct the graph
9//!   from scratch.
10//! * **cache** – serialise the graph to a compressed binary format using
11//!   [`icb_graph::cache`] so that subsequent runs can skip parsing
12//!   entirely.
13//! * **load** – restore a previously cached graph, automatically cleaning
14//!   up node names if needed (e.g. when the cache was created before the
15//!   USR‑to‑display‑name conversion was introduced).
16//!
17//! # Fact filtering
18//!
19//! Parsing a C++ project yields a large number of facts – not only
20//! functions and classes but also local variables, parameters, and
21//! intermediate AST scaffolding.  Only a small subset of these facts is
22//! needed for the call graph:
23//!
24//! * [`NodeKind::Function`] – callable entities,
25//! * [`NodeKind::Class`] – type containers,
26//! * [`NodeKind::CallSite`] – edges between callers and callees.
27//!
28//! All other facts (`Variable`, `Parameter`, …) are **discarded** before
29//! the graph is built.  This reduces the number of nodes by a factor of
30//! 10–100× and keeps both construction time and memory footprint
31//! predictable, even for large projects.
32//!
33//! # Name normalisation
34//!
35//! Clang emits Unified Symbol Resolution (USR) strings as unique
36//! identifiers.  The [`display_name`] module converts these strings into
37//! human‑readable names (e.g. `c:@F@main#` → `main`).  Normalisation
38//! happens in two places:
39//!
40//! * right after a new graph is built,
41//! * immediately after a graph is loaded from cache (the updated graph is
42//!   written back to the cache so the conversion is a one‑time cost).
43//!
44//! # Parallelism
45//!
46//! The parser itself processes translation units in parallel (see
47//! [`icb_clang::project`]).  Graph construction is intentionally
48//! single‑threaded – [`GraphBuilder::merge`] fuses per‑file sub‑graphs
49//! sequentially, which avoids lock contention on the central
50//! [`petgraph::StableGraph`].
51

59/// Builds a new graph or loads it from the specified cache file.
60///
61/// The resulting graph has **readable names** on every node: any Clang USR
62/// identifiers are automatically converted via
63/// [`display_name::readable_name`].
64///
65/// When loading from an existing cache, names are cleaned before returning
66/// the graph; if any names are changed, the cache is updated so that
67/// subsequent loads are instant.
68///
69/// # Arguments
70///
71/// * `project` - Path to the project root or a single file.
72/// * `language` - Programming language of the source files.
73/// * `compile_commands` - Optional path to a `compile_commands.json`.
74/// * `cpp_std` - C++ standard version to pass to the parser.
75/// * `cache_path` - Path to a cache file for faster reloading.
76/// * `no_system_headers` - Whether to exclude system header nodes.
77/// * `max_depth` - Maximum directory depth when scanning for files.
78///
79/// # Errors
80///
81/// Returns an error if the graph cannot be built or loaded.
82pub fn build_or_load_graph(
83    project: &Path,
84    language: &str,
85    compile_commands: Option<&PathBuf>,
86    cpp_std: &str,
87    cache_path: Option<&PathBuf>,
88    no_system_headers: bool,
89    max_depth: Option<usize>,
90) -> anyhow::Result<CodePropertyGraph> {
91    let lang = parse_language(language)?;
92
93    // Try loading from cache first
94    if let Some(cache_file) = cache_path {
95        if cache_file.exists() {
96            log::info!("Loading graph from cache {:?}", cache_file);
97            if let Ok(mut g) = cache::load_graph(cache_file) {
98                // Normalise names even for cached graphs
99                cleanup_node_names(&mut g);
100                // Persist the cleaned version so it's ready next time
101                if let Err(e) = cache::save_graph(&g, cache_file) {
102                    log::warn!("Failed to update cache with clean names: {}", e);
103                } else {
104                    log::info!("Cache updated with clean names");
105                }
106                return Ok(g);
107            }
108        }
109    }
110
111    let manager = icb_parser::manager::ParserManager::new();
112    let allow_system = !no_system_headers;
113    let file_facts = if lang == Language::Cpp {
114        if let Some(cdb) = compile_commands {
115            let cdb = cdb.canonicalize()?;
116            let base_dir = cdb.parent().unwrap_or(Path::new("."));
117            icb_clang::project::parse_project(&cdb, base_dir, true, allow_system)?
118        } else if project.is_file() {
119            let source = std::fs::read_to_string(project)?;
120            let args = vec![format!("-std={}", cpp_std)];
121            let facts = icb_clang::parser::parse_cpp_file(
122                &source,
123                &args,
124                Some(project.to_str().unwrap_or("unknown")),
125                allow_system,
126            )?;
127            vec![(
128                project
129                    .file_name()
130                    .unwrap_or_default()
131                    .to_string_lossy()
132                    .into_owned(),
133                facts,
134            )]
135        } else {
136            let args = vec![format!("-std={}", cpp_std)];
137            icb_clang::project::parse_directory(project, &args, true, max_depth, allow_system)?
138        }
139    } else if project.is_dir() {
140        manager.parse_directory(lang, project)?
141    } else {
142        let source = std::fs::read_to_string(project)?;
143        let facts = if lang == Language::Cpp {
144            let args = vec![format!("-std={}", cpp_std)];
145            icb_clang::parser::parse_cpp_file(
146                &source,
147                &args,
148                Some(project.to_str().unwrap_or("unknown")),
149                allow_system,
150            )?
151        } else {
152            manager.parse_file(lang, &source)?
153        };
154        vec![(
155            project
156                .file_name()
157                .unwrap_or_default()
158                .to_string_lossy()
159                .into_owned(),
160            facts,
161        )]
162    };
163
164    let mut builder = icb_graph::builder::GraphBuilder::new();
165    for (_, facts) in file_facts {
166        // Keep only the node kinds that form the call graph.
167        let filtered: Vec<_> = facts
168            .into_iter()
169            .filter(|f| {
170                matches!(
171                    f.kind,
172                    icb_common::NodeKind::Function
173                        | icb_common::NodeKind::Class
174                        | icb_common::NodeKind::CallSite
175                )
176            })
177            .collect();
178        let mut local = icb_graph::builder::GraphBuilder::new();
179        local.ingest_file_facts(&filtered);
180        builder.merge(local);
181    }
182    builder.resolve_calls();
183
184    let mut cpg = builder.cpg;
185
186    // Convert USR‑based names to readable display names before the graph is
187    // consumed by analytics or the API.
188    cleanup_node_names(&mut cpg);
189
190    if let Some(cache_file) = cache_path {
191        if let Err(e) = cache::save_graph(&cpg, cache_file) {
192            log::warn!("Failed to save cache: {}", e);
193        }
194    }
195    Ok(cpg)
196}

198/// Walks all graph nodes and replaces USR‑encoded names with their
199/// human‑readable equivalents.
200fn cleanup_node_names(cpg: &mut CodePropertyGraph) {
201    for node in cpg.graph.node_weights_mut() {
202        // Clean the primary display name
203        if let Some(ref name) = node.name {
204            let cleaned = display_name::readable_name(name);
205            if cleaned != *name {
206                node.name = Some(cleaned);
207            }
208        }
209
210        // For functions and classes, also clean the `usr` field if it
211        // appears to be a raw USR (starts with "c:").  This makes the
212        // field consistent with the display name used in the UI.
213        if node.kind == icb_common::NodeKind::Function || node.kind == icb_common::NodeKind::Class {
214            if let Some(ref usr) = node.usr {
215                if usr.starts_with("c:") {
216                    let cleaned = display_name::readable_name(usr);
217                    if cleaned != *usr {
218                        node.usr = Some(cleaned);
219                    }
220                }
221            }
222        }
223    }
224}

226fn parse_language(s: &str) -> anyhow::Result<Language> {
227    match s {
228        "python" => Ok(Language::Python),
229        "rust" => Ok(Language::Rust),
230        "javascript" => Ok(Language::JavaScript),
231        "cpp" | "c++" => Ok(Language::Cpp),
232        _ => anyhow::bail!("Unsupported language: {}", s),
233    }
