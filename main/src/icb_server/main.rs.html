1//! # icb-server
2//!
3//! Web server for ICB graph visualization and analytics.
4//!
5//! Serves static files from `web/` and exposes a REST API to explore
6//! the Code Property Graph, retrieve node details, and collect project
7//! metrics.
8
9use actix_files::Files;
10use actix_web::{web, App, HttpServer};
11use clap::Parser;
12use std::path::PathBuf;
13use std::sync::Mutex;
14
15mod analytics;
16mod graph_builder;
17mod routes;
18
19#[derive(Parser)]
20#[command(name = "icb-server")]
21#[command(about = "ICB web server for graph visualization")]
22pub struct Cli {
23    #[arg(short, long)]
24    pub project: PathBuf,
25
26    #[arg(short, long, default_value = "python")]
27    pub language: String,
28
29    #[arg(long)]
30    pub compile_commands: Option<PathBuf>,
31
32    #[arg(long, default_value = "c++17")]
33    pub cpp_std: String,
34
35    #[arg(long)]
36    pub cache: Option<PathBuf>,
37
38    #[arg(short = 'P', long, default_value = "8080")]
39    pub port: u16,
40
41    #[arg(long, default_value = "web")]
42    pub static_dir: PathBuf,
43
44    #[arg(long)]
45    pub no_system_headers: bool,
46
47    #[arg(long)]
48    pub max_depth: Option<usize>,
49}
50
51#[actix_web::main]
52async fn main() -> anyhow::Result<()> {
53    env_logger::init();
54    let args = Cli::parse();
55
56    let graph = graph_builder::build_or_load_graph(
57        &args.project,
58        &args.language,
59        args.compile_commands.as_ref(),
60        &args.cpp_std,
61        args.cache.as_ref(),
62        args.no_system_headers,
63        args.max_depth,
64    )?;
65
66    let graph_data = web::Data::new(Mutex::new(graph));
67    let static_dir = args.static_dir.canonicalize().unwrap_or(args.static_dir);
68    if !static_dir.exists() {
69        eprintln!("Warning: static directory {:?} does not exist", static_dir);
70    }
71    println!("Serving static files from {:?}", static_dir);
72    println!("Open http://localhost:{}", args.port);
73
74    HttpServer::new(move || {
75        App::new()
76            .app_data(graph_data.clone())
77            .configure(routes::configure)
78            .service(
79                Files::new("/", static_dir.clone())
80                    .index_file("index.html")
81                    .prefer_utf8(true),
82            )
83    })
84    .bind(("127.0.0.1", args.port))?
85    .run()
86    .await?;
87
88    Ok(())
89}
