11/// Returns `true` if the cursor is located in a system header.
12fn is_in_system_header(cursor: CXCursor) -> bool {
13    unsafe { clang_Location_isInSystemHeader(clang_getCursorLocation(cursor)) != 0 }
14}

16/// Returns the file name that contains the given cursor.
17fn cursor_file(cursor: CXCursor) -> Option<String> {
18    unsafe {
19        let loc = clang_getCursorLocation(cursor);
20        let mut file: CXFile = ptr::null_mut();
21        clang_getFileLocation(
22            loc,
23            &mut file,
24            ptr::null_mut(),
25            ptr::null_mut(),
26            ptr::null_mut(),
27        );
28        if file.is_null() {
29            return None;
30        }
31        let name = clang_getFileName(file);
32        let cstr = clang_getCString(name);
33        if cstr.is_null() {
34            None
35        } else {
36            Some(
37                std::ffi::CStr::from_ptr(cstr)
38                    .to_string_lossy()
39                    .into_owned(),
40            )
41        }
42    }
43}

45/// Parse C/C++ source code using Clang and return a flat list of facts.
46///
47/// The source is written to a temporary file and Clang is invoked with `args`.
48/// If `allow_system` is `false`, nodes originating from system headers are
49/// excluded from the result.
50///
51/// # Errors
52///
53/// Returns [`IcbError::Parse`] if the Clang API cannot be initialised or
54/// the source cannot be parsed.
55pub fn parse_cpp_file(
56    source: &str,
57    args: &[String],
58    _file_name: Option<&str>,
59    allow_system: bool,
60) -> Result<Vec<RawNode>, IcbError> {
61    let index = unsafe { clang_createIndex(0, 0) };
62    if index.is_null() {
63        return Err(IcbError::Parse("failed to create Clang index".into()));
64    }
65
66    let temp_file = Builder::new()
67        .suffix(".cpp")
68        .tempfile()
69        .map_err(|e| IcbError::Parse(format!("tempfile error: {}", e)))?;
70    std::fs::write(temp_file.path(), source).map_err(IcbError::Io)?;
71    let filename = temp_file
72        .path()
73        .to_str()
74        .ok_or_else(|| IcbError::Parse("non-UTF8 temp path".into()))?;
75    let filename_c = CString::new(filename).unwrap();
76
77    let arg_ptrs: Vec<*const i8> = args
78        .iter()
79        .map(|a| CString::new(a.as_str()).unwrap().into_raw() as *const i8)
80        .collect();
81    let mut tu: CXTranslationUnit = ptr::null_mut();
82
83    let error = unsafe {
84        clang_parseTranslationUnit2(
85            index,
86            filename_c.as_ptr(),
87            arg_ptrs.as_ptr(),
88            args.len() as i32,
89            ptr::null_mut(),
90            0,
91            CXTranslationUnit_None,
92            &mut tu,
93        )
94    };
95
96    for &cstr_ptr in &arg_ptrs {
97        unsafe {
98            let _ = CString::from_raw(cstr_ptr as *mut i8);
99        }
100    }
101
102    if error != CXError_Success {
103        unsafe { clang_disposeIndex(index) };
104        return Err(IcbError::Parse(format!(
105            "failed to parse translation unit, error code {:?}",
106            error
107        )));
108    }
109
110    let cursor = unsafe { clang_getTranslationUnitCursor(tu) };
111    let mut nodes = Vec::new();
112    visit_children(cursor, &mut nodes, None, false, allow_system);
113
114    unsafe {
115        clang_disposeTranslationUnit(tu);
116        clang_disposeIndex(index);
117    }
118
119    Ok(nodes)
120}

122struct VisitorContext<'a> {
123    nodes: &'a mut Vec<RawNode>,
124    latest_parent: Option<usize>,
125    in_system: bool,
126    allow_system: bool,
127}
128
129fn visit_children(
130    cursor: CXCursor,
131    nodes: &mut Vec<RawNode>,
132    parent_idx: Option<usize>,
133    in_system: bool,
134    allow_system: bool,
135) -> Option<usize> {
136    let is_sys = is_in_system_header(cursor);
137
138    // If system headers are disabled, skip the whole sub‑tree.
139    if !allow_system && is_sys {
140        return parent_idx;
141    }
142
143    // If we are already inside a system header, do not recurse further.
144    if in_system && is_sys {
145        return parent_idx;
146    }
147
148    let kind = unsafe { clang_getCursorKind(cursor) };
149    let (node_kind, name, usr) = match kind {
150        CXCursor_FunctionDecl => (
151            NodeKind::Function,
152            Some(cursor_spelling(cursor)),
153            Some(cursor_usr(cursor)),
154        ),
155        CXCursor_CXXMethod => (
156            NodeKind::Function,
157            Some(cursor_spelling(cursor)),
158            Some(cursor_usr(cursor)),
159        ),
160        CXCursor_ClassDecl | CXCursor_StructDecl => (
161            NodeKind::Class,
162            Some(cursor_spelling(cursor)),
163            Some(cursor_usr(cursor)),
164        ),
165        CXCursor_CallExpr => {
166            let referenced = unsafe { clang_getCursorReferenced(cursor) };
167            let name = if referenced.kind == CXCursor_InvalidFile {
168                cursor_spelling(cursor)
169            } else {
170                cursor_spelling(referenced)
171            };
172            (NodeKind::CallSite, Some(name), None)
173        }
174        CXCursor_VarDecl => (NodeKind::Variable, Some(cursor_spelling(cursor)), None),
175        CXCursor_ParmDecl => (NodeKind::Parameter, Some(cursor_spelling(cursor)), None),
176        _ => {
177            let mut ctx = VisitorContext {
178                nodes,
179                latest_parent: parent_idx,
180                in_system,
181                allow_system,
182            };
183            let ctx_ptr: *mut c_void = &mut ctx as *mut _ as *mut c_void;
184            unsafe {
185                clang_visitChildren(cursor, visitor_callback, ctx_ptr);
186            }
187            return ctx.latest_parent;
188        }
189    };
190
191    let (start_line, start_col, end_line, end_col) = cursor_location(cursor);
192
193    let idx = nodes.len();
194    nodes.push(RawNode {
195        language: Language::Cpp,
196        kind: node_kind,
197        name,
198        usr,
199        start_line,
200        start_col,
201        end_line,
202        end_col,
203        children: Vec::new(),
204        source_file: cursor_file(cursor),
205    });
206
207    if let Some(pidx) = parent_idx {
208        nodes[pidx].children.push(idx);
209    }
210
211    let next_in_system = in_system || is_sys;
212
213    let mut ctx = VisitorContext {
214        nodes,
215        latest_parent: Some(idx),
216        in_system: next_in_system,
217        allow_system,
218    };
219    let ctx_ptr: *mut c_void = &mut ctx as *mut _ as *mut c_void;
220    unsafe {
221        clang_visitChildren(cursor, visitor_callback, ctx_ptr);
222    }
223
224    ctx.latest_parent
225}

227extern "C" fn visitor_callback(
228    cursor: CXCursor,
229    _parent: CXCursor,
230    client_data: CXClientData,
231) -> CXChildVisitResult {
232    let ctx: &mut VisitorContext = unsafe { &mut *(client_data as *mut VisitorContext) };
233    ctx.latest_parent = visit_children(
234        cursor,
235        ctx.nodes,
236        ctx.latest_parent,
237        ctx.in_system,
238        ctx.allow_system,
239    );
240    CXChildVisit_Continue
241}

243fn cursor_location(cursor: CXCursor) -> (usize, usize, usize, usize) {
244    let range = unsafe { clang_getCursorExtent(cursor) };
245    let start = unsafe { clang_getRangeStart(range) };
246    let end = unsafe { clang_getRangeEnd(range) };
247
248    let mut line: c_uint = 0;
249    let mut column: c_uint = 0;
250
251    unsafe {
252        clang_getPresumedLocation(start, ptr::null_mut(), &mut line, &mut column);
253    }
254    let s_line = line as usize;
255    let s_col = column as usize;
256
257    unsafe {
258        clang_getPresumedLocation(end, ptr::null_mut(), &mut line, &mut column);
259    }
260    let e_line = line as usize;
261    let e_col = column as usize;
262
263    (s_line, s_col, e_line, e_col)
264}

266fn cursor_spelling(cursor: CXCursor) -> String {
267    unsafe {
268        let cxstring = clang_getCursorSpelling(cursor);
269        let s = clang_getCString(cxstring);
270        let result = if s.is_null() {
271            String::new()
272        } else {
273            std::ffi::CStr::from_ptr(s).to_string_lossy().into_owned()
274        };
275        clang_disposeString(cxstring);
276        result
277    }
278}
