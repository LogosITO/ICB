17#[derive(clap::Subcommand)]
18enum Command {
19    /// Build Code Property Graph and show basic statistics.
20    Analyze {
21        path: PathBuf,
22        #[arg(short, long)]
23        language: String,
24        #[arg(long)]
25        compile_commands: Option<PathBuf>,
26        #[arg(long, default_value = "c++17")]
27        cpp_std: String,
28        #[arg(long)]
29        cache: Option<PathBuf>,
30    },
31    /// Run queries on a project directory.
32    Query {
33        project: PathBuf,
34        #[arg(short, long, default_value = "python")]
35        language: String,
36        #[arg(long)]
37        compile_commands: Option<PathBuf>,
38        #[arg(long, default_value = "c++17")]
39        cpp_std: String,
40        #[arg(long)]
41        functions: bool,
42        #[arg(long)]
43        callers: Option<String>,
44        #[arg(long)]
45        callees: Option<String>,
46        #[arg(long)]
47        unused: bool,
48        #[arg(long)]
49        dot: bool,
50        #[arg(long)]
51        cycles: bool,
52        #[arg(long)]
53        dead_code: bool,
54        #[arg(long, default_value = "main", requires = "dead_code")]
55        entries: String,
56        #[arg(long)]
57        complexity: bool,
58        #[arg(long, default_value = "20", requires = "complexity")]
59        threshold: usize,
60        #[arg(long)]
61        cache: Option<PathBuf>,
62    },
63}

65fn main() -> anyhow::Result<()> {
66    env_logger::init();
67    let cli = Cli::parse();
68    let manager = ParserManager::new();
69
70    match cli.command {
71        Command::Analyze {
72            path,
73            language,
74            compile_commands,
75            cpp_std,
76            cache: cache_path,
77        } => {
78            let lang = parse_language(&language)?;
79            let (cpg, _files) = build_or_load_graph(
80                &manager,
81                lang,
82                &path,
83                compile_commands.as_deref(),
84                &cpp_std,
85                cache_path.as_deref(),
86                true,
87            )?;
88            println!(
89                "Graph: {} nodes, {} edges",
90                cpg.node_count(),
91                cpg.edge_count()
92            );
93        }
94        Command::Query {
95            project,
96            language,
97            compile_commands,
98            cpp_std,
99            functions,
100            callers,
101            callees,
102            unused,
103            dot,
104            cycles,
105            dead_code,
106            entries,
107            complexity,
108            threshold,
109            cache: cache_path,
110        } => {
111            let lang = parse_language(&language)?;
112            let (cpg, _files) = build_or_load_graph(
113                &manager,
114                lang,
115                &project,
116                compile_commands.as_deref(),
117                &cpp_std,
118                cache_path.as_deref(),
119                false,
120            )?;
121
122            if functions {
123                print_functions(&cpg);
124            }
125            if let Some(target) = callers {
126                print_callers(&cpg, &target);
127            }
128            if let Some(target) = callees {
129                print_callees(&cpg, &target);
130            }
131            if unused {
132                print_unused(&cpg);
133            }
134            if dot {
135                println!("{}", visualizer::export_call_dot(&cpg));
136            }
137            if cycles {
138                print_cycles(&cpg);
139            }
140            if dead_code {
141                let entry_list: Vec<String> =
142                    entries.split(',').map(|s| s.trim().to_string()).collect();
143                print_dead_code(&cpg, &entry_list);
144            }
145            if complexity {
146                print_complexity(&cpg, threshold);
147            }
148        }
149    }
150
151    Ok(())
152}

154fn parse_language(s: &str) -> anyhow::Result<Language> {
155    match s {
156        "python" => Ok(Language::Python),
157        "rust" => Ok(Language::Rust),
158        "javascript" => Ok(Language::JavaScript),
159        "cpp" | "c++" => Ok(Language::Cpp),
160        _ => anyhow::bail!("Unsupported language: {}", s),
161    }
162}

164fn build_or_load_graph(
165    manager: &ParserManager,
166    lang: Language,
167    path: &Path,
168    compile_commands: Option<&Path>,
169    cpp_std: &str,
170    cache_path: Option<&Path>,
171    show_progress: bool,
172) -> anyhow::Result<(icb_graph::graph::CodePropertyGraph, usize)> {
173    if let Some(cache_file) = cache_path {
174        if cache_file.exists() {
175            log::info!("Loading cached graph from {:?}", cache_file);
176            match cache::load_graph(cache_file) {
177                Ok(cpg) => {
178                    log::info!(
179                        "Using cached graph ({} nodes, {} edges)",
180                        cpg.node_count(),
181                        cpg.edge_count()
182                    );
183                    return Ok((cpg, 0));
184                }
185                Err(e) => log::warn!("Failed to load cache: {}", e),
186            }
187        }
188    }
189
190    let (cpg, files_count) = build_project_graph(
191        manager,
192        lang,
193        path,
194        compile_commands,
195        cpp_std,
196        show_progress,
197    )?;
198
199    if let Some(cache_file) = cache_path {
200        if let Err(e) = cache::save_graph(&cpg, cache_file) {
201            log::warn!("Failed to save cache: {}", e);
202        } else {
203            log::info!("Graph cached to {:?}", cache_file);
204        }
205    }
206
207    Ok((cpg, files_count))
208}

210fn build_project_graph(
211    manager: &ParserManager,
212    lang: Language,
213    path: &Path,
214    compile_commands: Option<&Path>,
215    cpp_std: &str,
216    show_progress: bool,
217) -> anyhow::Result<(icb_graph::graph::CodePropertyGraph, usize)> {
218    let file_facts: Vec<(String, Vec<icb_parser::facts::RawNode>)> = if lang == Language::Cpp {
219        if let Some(cdb) = compile_commands {
220            let cdb = cdb.canonicalize()?;
221            let base_dir = cdb.parent().unwrap_or(Path::new("."));
222            icb_clang::project::parse_project(&cdb, base_dir, true, true)?
223        } else if path.is_file() {
224            let source = std::fs::read_to_string(path)?;
225            let args = vec![format!("-std={}", cpp_std)];
226            let facts = icb_clang::parser::parse_cpp_file(
227                &source,
228                &args,
229                Some(
230                    path.file_name()
231                        .unwrap_or_default()
232                        .to_str()
233                        .unwrap_or("unknown"),
234                ),
235                true,
236            )?;
237            vec![(
238                path.file_name()
239                    .unwrap_or_default()
240                    .to_string_lossy()
241                    .into_owned(),
242                facts,
243            )]
244        } else {
245            let args = vec![format!("-std={}", cpp_std)];
246            icb_clang::project::parse_directory(path, &args, true, None, true)?
247        }
248    } else if path.is_dir() {
249        manager.parse_directory(lang, path)?
250    } else {
251        let source = std::fs::read_to_string(path)?;
252        let facts = if lang == Language::Cpp {
253            let args = vec![format!("-std={}", cpp_std)];
254            icb_clang::parser::parse_cpp_file(
255                &source,
256                &args,
257                Some(
258                    path.file_name()
259                        .unwrap_or_default()
260                        .to_str()
261                        .unwrap_or("unknown"),
262                ),
263                true,
264            )?
265        } else {
266            manager.parse_file(lang, &source)?
267        };
268        vec![(
269            path.file_name()
270                .unwrap_or_default()
271                .to_string_lossy()
272                .into_owned(),
273            facts,
274        )]
275    };
276
277    let files_count = file_facts.len();
278    if show_progress {
279        println!("Parsed {} files", files_count);
280    }
281
282    let mut global_builder = GraphBuilder::new();
283    for (_, facts) in file_facts {
284        let mut local = GraphBuilder::new();
285        local.ingest_file_facts(&facts);
286        global_builder.merge(local);
287    }
288    global_builder.resolve_calls();
289
290    Ok((global_builder.cpg, files_count))
291}

293fn print_functions(cpg: &icb_graph::graph::CodePropertyGraph) {
294    let funcs = query::find_by_kind(cpg, icb_common::NodeKind::Function);
295    println!("Functions ({})", funcs.len());
296    for f in &funcs {
297        println!(
298            "  {} (line {})",
299            f.name.as_deref().unwrap_or("?"),
300            f.start_line
301        );
302    }
303}

305fn print_callers(cpg: &icb_graph::graph::CodePropertyGraph, target: &str) {

307    println!("Callers of '{}' ({})", target, callers.len());
308    for (caller, _) in &callers {
309        println!(
310            "  {} (line {})",
311            caller.name.as_deref().unwrap_or("?"),
312            caller.start_line
313        );
314    }
315}

341fn print_cycles(cpg: &icb_graph::graph::CodePropertyGraph) {
342    let cycles = analysis::detect_call_cycles(cpg);
343    println!("Call cycles ({})", cycles.len());
344    for cycle in &cycles {
345        println!("  Length {}: {}", cycle.length, cycle.functions.join(", "));
346    }

349fn print_dead_code(cpg: &icb_graph::graph::CodePropertyGraph, entries: &[String]) {
350    let dead = analysis::detect_dead_code(cpg, entries);
351    println!("Dead code from entries {:?} ({})", entries, dead.len());
352    for f in &dead {
353        println!(
354            "  {} (line {})",
355            f.name.as_deref().unwrap_or("?"),
356            f.start_line
357        );
358    }
359}

361fn print_complexity(cpg: &icb_graph::graph::CodePropertyGraph, threshold: usize) {
362    let complex = analysis::detect_complex_functions(cpg, threshold);
363    println!(
364        "Complex functions (threshold {}): {}",
365        threshold,
366        complex.len()
367    );
368    for report in &complex {
369        println!(
370            "  {} (AST nodes: {}, line {})",
371            report.function_name, report.ast_node_count, report.start_line
372        );
373    }
