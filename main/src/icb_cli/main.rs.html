101fn main() -> anyhow::Result<()> {
102    env_logger::init();
103    let cli = Cli::parse();
104    let manager = ParserManager::new();
105
106    match cli.command {
107        Command::Analyze {
108            path,
109            language,
110            compile_commands,
111            cpp_std,
112            cache: cache_path,
113            no_system_headers,
114        } => {
115            let lang = parse_language(&language)?;
116            let opts = BuildOptions {
117                manager: &manager,
118                lang,
119                path: &path,
120                compile_commands: compile_commands.as_deref(),
121                cpp_std: &cpp_std,
122                cache_path: cache_path.as_deref(),
123                show_progress: true,
124                no_system_headers,
125            };
126            let (cpg, _) = build_or_load_graph(opts)?;
127            println!(
128                "Graph: {} nodes, {} edges",
129                cpg.node_count(),
130                cpg.edge_count()
131            );
132        }
133        Command::Query {
134            project,
135            language,
136            compile_commands,
137            cpp_std,
138            functions,
139            callers,
140            callees,
141            unused,
142            dot,
143            cycles,
144            dead_code,
145            entries,
146            complexity,
147            threshold,
148            cache: cache_path,
149            no_system_headers,
150        } => {
151            let lang = parse_language(&language)?;
152            let opts = BuildOptions {
153                manager: &manager,
154                lang,
155                path: &project,
156                compile_commands: compile_commands.as_deref(),
157                cpp_std: &cpp_std,
158                cache_path: cache_path.as_deref(),
159                show_progress: false,
160                no_system_headers,
161            };
162            let (cpg, _) = build_or_load_graph(opts)?;
163            if functions {
164                print_functions(&cpg);
165            }
166            if let Some(target) = callers {
167                print_callers(&cpg, &target);
168            }
169            if let Some(target) = callees {
170                print_callees(&cpg, &target);
171            }
172            if unused {
173                print_unused(&cpg);
174            }
175            if dot {
176                println!("{}", visualizer::export_call_dot(&cpg));
177            }
178            if cycles {
179                print_cycles(&cpg);
180            }
181            if dead_code {
182                let entry_list: Vec<String> =
183                    entries.split(',').map(|s| s.trim().to_string()).collect();
184                print_dead_code(&cpg, &entry_list);
185            }
186            if complexity {
187                print_complexity(&cpg, threshold);
188            }
189        }
190        Command::Report {
191            project,
192            language,
193            compile_commands,
194            cpp_std,
195            cache: cache_path,
196            no_system_headers,
197            output,
198        } => {
199            let lang = parse_language(&language)?;
200            let opts = BuildOptions {
201                manager: &manager,
202                lang,
203                path: &project,
204                compile_commands: compile_commands.as_deref(),
205                cpp_std: &cpp_std,
206                cache_path: cache_path.as_deref(),
207                show_progress: true,
208                no_system_headers,
209            };
210            let (cpg, _) = build_or_load_graph(opts)?;
211            let html = icb_report::report::generate_report(&cpg, &project.display().to_string())?;
212            fs::write(&output, html)?;
213            println!("Report written to {:?}", output);
214        }
215        Command::Diff {
216            old_project,
217            new_project,
218            language,
219            compile_commands,
220            cpp_std,
221            cache: cache_path,
222            no_system_headers,
223            output,
224        } => {
225            let lang = parse_language(&language)?;
226            let opts_old = BuildOptions {
227                manager: &manager,
228                lang,
229                path: &old_project,
230                compile_commands: compile_commands.as_deref(),
231                cpp_std: &cpp_std,
232                cache_path: cache_path.as_deref(),
233                show_progress: true,
234                no_system_headers,
235            };
236            let (old_cpg, _) = build_or_load_graph(opts_old)?;
237            let opts_new = BuildOptions {
238                manager: &manager,
239                lang,
240                path: &new_project,
241                compile_commands: compile_commands.as_deref(),
242                cpp_std: &cpp_std,
243                cache_path: cache_path.as_deref(),
244                show_progress: true,
245                no_system_headers,
246            };
247            let (new_cpg, _) = build_or_load_graph(opts_new)?;
248            let html = icb_report::diff::generate_diff(&old_cpg, &new_cpg, "Project")?;
249            fs::write(&output, html)?;
250            println!("Diff written to {:?}", output);
251        }
252    }
253    Ok(())
254}

256struct BuildOptions<'a> {
257    manager: &'a ParserManager,

259    path: &'a Path,
260    compile_commands: Option<&'a Path>,
261    cpp_std: &'a str,
262    cache_path: Option<&'a Path>,
263    show_progress: bool,

267fn parse_language(s: &str) -> anyhow::Result<Language> {
268    match s {
269        "python" => Ok(Language::Python),
270        "rust" => Ok(Language::Rust),
271        "javascript" => Ok(Language::JavaScript),
272        "cpp" | "c++" => Ok(Language::Cpp),
273        _ => anyhow::bail!("Unsupported language: {}", s),
274    }
275}

277fn build_or_load_graph(
278    opts: BuildOptions,
279) -> anyhow::Result<(icb_graph::graph::CodePropertyGraph, usize)> {
280    if let Some(cache_file) = opts.cache_path {
281        if cache_file.exists() {
282            if let Ok(cpg) = cache::load_graph(cache_file) {
283                return Ok((cpg, 0));
284            }
285        }
286    }
287    let file_facts = build_file_facts(
288        opts.manager,
289        opts.lang,
290        opts.path,
291        opts.compile_commands,
292        opts.cpp_std,
293        opts.no_system_headers,
294    )?;
295    let count = file_facts.len();
296    if opts.show_progress {
297        println!("Parsed {} files", count);
298    }
299    let mut builder = GraphBuilder::new();
300    for (_, facts) in file_facts {
301        let mut local = GraphBuilder::new();
302        local.ingest_file_facts(&facts);
303        builder.merge(local);
304    }
305    builder.resolve_calls();
306    let cpg = builder.cpg;
307    if let Some(cache_file) = opts.cache_path {
308        if let Err(e) = cache::save_graph(&cpg, cache_file) {
309            log::warn!("Failed to save cache: {}", e);
310        }
311    }
312    Ok((cpg, count))
313}

315fn build_file_facts(
316    manager: &ParserManager,
317    lang: Language,
318    path: &Path,
319    compile_commands: Option<&Path>,
320    cpp_std: &str,
321    no_system_headers: bool,
322) -> anyhow::Result<Vec<(String, Vec<icb_parser::facts::RawNode>)>> {
323    let allow_system = !no_system_headers;
324    if lang == Language::Cpp {
325        if let Some(cdb) = compile_commands {
326            let cdb = cdb.canonicalize()?;
327            let base_dir = cdb.parent().unwrap_or(Path::new("."));
328            Ok(icb_clang::project::parse_project(
329                &cdb,
330                base_dir,
331                true,
332                allow_system,
333            )?)
334        } else if path.is_file() {
335            let source = std::fs::read_to_string(path)?;
336            let args = vec![format!("-std={}", cpp_std)];
337            let facts = icb_clang::parser::parse_cpp_file(
338                &source,
339                &args,
340                Some(path.to_str().unwrap()),
341                allow_system,
342            )?;
343            Ok(vec![(
344                path.file_name()
345                    .unwrap_or_default()
346                    .to_string_lossy()
347                    .into_owned(),
348                facts,
349            )])
350        } else {
351            Ok(icb_clang::project::parse_directory(
352                path,
353                &[format!("-std={}", cpp_std)],
354                true,
355                None,
356                allow_system,
357            )?)
358        }
359    } else if path.is_dir() {
360        Ok(manager.parse_directory(lang, path)?)
361    } else {
362        let source = std::fs::read_to_string(path)?;
363        let facts = manager.parse_file(lang, &source)?;
364        Ok(vec![(
365            path.file_name()
366                .unwrap_or_default()
367                .to_string_lossy()
368                .into_owned(),
369            facts,
370        )])
371    }
372}

374fn print_functions(cpg: &icb_graph::graph::CodePropertyGraph) {
375    let funcs = query::find_by_kind(cpg, icb_common::NodeKind::Function);
376    println!("Functions ({})", funcs.len());
377    for f in &funcs {
378        println!(
379            "  {} (line {})",
380            f.name.as_deref().unwrap_or("?"),
381            f.start_line
382        );
383    }
384}
385fn print_callers(cpg: &icb_graph::graph::CodePropertyGraph, target: &str) {
386    let callers = query::callers_of(cpg, target);
387    println!("Callers of '{}' ({})", target, callers.len());
388    for (caller, _) in &callers {
389        println!(
390            "  {} (line {})",
391            caller.name.as_deref().unwrap_or("?"),
392            caller.start_line
393        );
394    }
395}
396fn print_callees(cpg: &icb_graph::graph::CodePropertyGraph, target: &str) {
397    let callees = query::callees_of(cpg, target);
398    println!("Callees of '{}' ({})", target, callees.len());
399    for (callee, _) in &callees {
400        println!(
401            "  {} (line {})",
402            callee.name.as_deref().unwrap_or("?"),
403            callee.start_line
404        );
405    }
406}
407fn print_unused(cpg: &icb_graph::graph::CodePropertyGraph) {
408    let unused = query::unused_functions(cpg);
409    println!("Unused functions ({})", unused.len());
410    for f in &unused {
411        println!(
412            "  {} (line {})",
413            f.name.as_deref().unwrap_or("?"),
414            f.start_line
415        );

418fn print_cycles(cpg: &icb_graph::graph::CodePropertyGraph) {
419    let cycles = analysis::detect_call_cycles(cpg);
420    println!("Call cycles ({})", cycles.len());
421    for cycle in &cycles {
422        println!("  Length {}: {}", cycle.length, cycle.functions.join(", "));
423    }
424}
425fn print_dead_code(cpg: &icb_graph::graph::CodePropertyGraph, entries: &[String]) {
426    let dead = analysis::detect_dead_code(cpg, entries);
427    println!("Dead code from entries {:?} ({})", entries, dead.len());
428    for f in &dead {
429        println!(
430            "  {} (line {})",
431            f.name.as_deref().unwrap_or("?"),
432            f.start_line
433        );
434    }
435}
436fn print_complexity(cpg: &icb_graph::graph::CodePropertyGraph, threshold: usize) {
437    let complex = analysis::detect_complex_functions(cpg, threshold);
438    println!(
439        "Complex functions (threshold {}): {}",
440        threshold,
441        complex.len()
442    );
443    for report in &complex {
444        println!(
445            "  {} (AST nodes: {}, line {})",
446            report.function_name, report.ast_node_count, report.start_line
447        );
448    }
