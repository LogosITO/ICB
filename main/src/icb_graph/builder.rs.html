6/// Incrementally builds a [`CodePropertyGraph`] from parser facts.
7///
8/// Supports ingesting facts from multiple files and merging local graphs
9/// (e.g., from parallel parsing) into a single global graph with symbol
10/// deduplication.
11///
12/// # Example
13///
14/// ```rust
15/// use icb_graph::builder::GraphBuilder;
16/// use icb_parser::facts::RawNode;
17/// use icb_common::{Language, NodeKind};
18///
19/// let mut builder = GraphBuilder::new();
20/// let facts = vec![RawNode {
21///     language: Language::Python,
22///     kind: NodeKind::Function,
23///     name: Some("main".into()),
24///     usr: None,
25///     start_line: 1,
26///     start_col: 0,
27///     end_line: 1,
28///     end_col: 5,
29///     children: vec![],
30/// }];
31/// builder.ingest_file_facts(&facts);
32/// assert_eq!(builder.cpg.node_count(), 1);
33/// ```
34#[derive(Default)]
35pub struct GraphBuilder {
36    /// The Code Property Graph under construction.
37    pub cpg: CodePropertyGraph,
38    /// Map from symbolic keys (USR or name) to graph node indices for
39    /// deduplication.
40    symbol_index: HashMap<String, NodeIndex>,
41}
42
43impl GraphBuilder {
44    /// Create an empty builder.
45    pub fn new() -> Self {
46        Self::default()
47    }
48
49    /// Ingest facts from a single file.
50    ///
51    /// This method may be called multiple times (even from different threads
52    /// after merging) — each call enriches the same graph.
53    pub fn ingest_file_facts(&mut self, facts: &[RawNode]) {
54        let mut map: HashMap<usize, NodeIndex> = HashMap::new();
55        for (i, raw) in facts.iter().enumerate() {
56            let usr: String = raw
57                .usr
58                .clone()
59                .or_else(|| raw.name.clone())
60                .unwrap_or_default();
61            let node_idx = if let Some(&existing) = self.symbol_index.get(&usr) {
62                existing
63            } else {
64                let idx = self.cpg.graph.add_node(Node {
65                    kind: raw.kind.clone(),
66                    name: raw.name.clone(),
67                    usr: Some(usr.clone()),
68                    start_line: raw.start_line,
69                    end_line: raw.end_line,
70                });
71                self.symbol_index.insert(usr, idx);
72                idx
73            };
74            map.insert(i, node_idx);
75        }
76
77        for (i, raw) in facts.iter().enumerate() {
78            let from_idx = map[&i];
79            for &child_raw_idx in &raw.children {
80                if let Some(&to_idx) = map.get(&child_raw_idx) {
81                    self.cpg.graph.add_edge(from_idx, to_idx, Edge::AstChild);
82                }
83            }
84        }
85    }
86
87    /// Merge another `GraphBuilder` into this one.
88    ///
89    /// All nodes from `other` are transferred to `self`. Nodes with the
90    /// same symbolic key (USR or name) that already exist in `self` are
91    /// reused, and edges are rewired accordingly. This is the main
92    /// mechanism to combine graphs from parallel file parsing.
93    pub fn merge(&mut self, other: GraphBuilder) {
94        // First, transfer all nodes from other, reusing existing ones
95        let mut node_map: HashMap<NodeIndex, NodeIndex> = HashMap::new();
96        for old_idx in other.cpg.graph.node_indices() {
97            let node = &other.cpg.graph[old_idx];
98            let usr = node.usr.clone().unwrap_or_default();
99            let new_idx = if let Some(&existing) = self.symbol_index.get(&usr) {
100                existing
101            } else {
102                let idx = self.cpg.graph.add_node(node.clone());
103                self.symbol_index.insert(usr, idx);
104                idx
105            };
106            node_map.insert(old_idx, new_idx);
107        }
108        // Transfer edges
109        for edge_ref in other.cpg.graph.edge_indices() {
110            let (source, target) = other.cpg.graph.edge_endpoints(edge_ref).unwrap();
111            let new_source = node_map[&source];
112            let new_target = node_map[&target];
113            self.cpg
114                .graph
115                .add_edge(new_source, new_target, other.cpg.graph[edge_ref].clone());
116        }
117    }
118}

120#[cfg(test)]
121mod tests {
122    use super::*;
123    use icb_common::{Language, NodeKind};
124
125    fn make_func_node(name: &str, line: usize) -> RawNode {
126        RawNode {
127            language: Language::Python,
128            kind: NodeKind::Function,
129            name: Some(name.into()),
130            usr: None,
131            start_line: line,
132            start_col: 0,
133            end_line: line,
134            end_col: 10,
135            children: vec![],
136        }
137    }
138
139    #[test]
140    fn test_deduplication_by_name() {
141        let mut builder = GraphBuilder::new();
142        let facts = vec![
143            make_func_node("foo", 1),
144            make_func_node("foo", 2), // duplicate
145        ];
146        builder.ingest_file_facts(&facts);
147        assert_eq!(builder.cpg.node_count(), 1);
148    }
149
150    #[test]
151    fn test_multiple_different_nodes() {
152        let mut builder = GraphBuilder::new();
153        let facts = vec![make_func_node("foo", 1), make_func_node("bar", 2)];
154        builder.ingest_file_facts(&facts);
155        assert_eq!(builder.cpg.node_count(), 2);
156    }
157
158    #[test]
159    fn test_merge_two_builders() {
160        let mut b1 = GraphBuilder::new();
161        b1.ingest_file_facts(&[make_func_node("foo", 1)]);
162        let mut b2 = GraphBuilder::new();
163        b2.ingest_file_facts(&[make_func_node("bar", 2), make_func_node("foo", 3)]); // duplicate foo
164        b1.merge(b2);
165        assert_eq!(b1.cpg.node_count(), 2); // foo, bar
166    }
